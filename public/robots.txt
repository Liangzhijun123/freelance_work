import React, { useState } from "react";

const Pop_out = ({ images }) => {
  const [slideIndex, setSlideIndex] = useState(0);

  const plusDivs = (n) => {
    const newIndex = slideIndex + n;
    if (newIndex >= 0 && newIndex < images.length) {
      setSlideIndex(newIndex);
    }
  };

  const showDivs = (index) => {
    return (
      <div className="slideshow-container">
        {images.map((image, idx) => (
          <div
            key={idx}
            className={idx === index ? "mySlides fade" : "mySlides"}
            style={{ display: idx === index ? "block" : "none" }}
          >
            <img src={image} alt={`Slide ${idx + 1}`} />
          </div>
        ))}
        <button className="prev group p-2" onClick={() => plusDivs(-1)}>
          <img
            src="/image/leftarrow.svg"
            alt="left arrow"
            className="block group-hover:hidden"
          />
          <img
            src="/image/blackleftarrow.svg"
            alt="black left arrow"
            className="hidden group-hover:block"
          />
        </button>
        <button className="next group p-2" onClick={() => plusDivs(1)}>
          <img
            src="/image/rightarrow.svg"
            alt="right arrow"
            className="block group-hover:hidden"
          />
          <img
            src="/image/blackrightarrow.svg"
            alt="black right arrow"
            className="hidden group-hover:block"
          />
        </button>
      </div>
    );
  };

  return showDivs(slideIndex);
};

export default Pop_out;
